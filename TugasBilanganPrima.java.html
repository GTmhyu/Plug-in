import java.util.Scanner;

public class TugasBilanganPrima {
    public static void Cetakjudul() {
        String Nim;
        String Nama;
        String Kelas;
        String Project;
        Nim = "21090109";
        Nama = "Ziidan Afkar Maula";
        Kelas = "1A";
        Project = "Simple method";

        System.out.println("Nim :" + Nim);
        System.out.println("Nama :" + Nama);
        System.out.println("Kelas :" + Kelas);
        System.out.println("Project :" + Project);
    }

    public static void Angka() {
        Scanner scan = new Scanner (System.in);
        System.out.print("Input bilangan : ");
        int bilangan = scan.nextInt();
        int n = 0;
        if (bilangan >= 2) {
            for (int x = 1; x <= bilangan; x++) {
                int m = bilangan % x;
                if (m == 0) {
                    n++;
                }
            }
            if (n == 2) {
                System.out.print(bilangan + " adalah bilangan prima");
            } else {
                System.out.print(bilangan + " adalah bilangan bukan prima ");
            }
        } else {
            System.out.print(bilangan + " adalah bilangan bukan prima");
        }
    }



    public static void main(String[] args) {
        Cetakjudul();
        Angka();

    }
}
